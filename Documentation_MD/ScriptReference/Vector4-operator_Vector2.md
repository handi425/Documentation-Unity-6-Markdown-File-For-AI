### Description

Converts a [Vector2](Vector2.html) to a Vector4.

Vector2s can be implicitly converted to Vector4 (z and w is set to zero in the
result).

    
    
    using UnityEngine;
    using System.Collections.Generic;  
      
    
    public class ExampleScript : [MonoBehaviour](MonoBehaviour.html)
    {
        void Start()
        {
            [Renderer](Renderer.html) renderer = GetComponent<[Renderer](Renderer.html)>();  
      
            // [Shader](Shader.html) vectors are always defined as [Vector4](Vector4.html).
            // The value here is converted to a [Vector4](Vector4.html) from a [Vector2](Vector2.html).
            renderer.material.SetVector("_SomeVariable", [Vector2.one](Vector2-one.html));
        }
    }
