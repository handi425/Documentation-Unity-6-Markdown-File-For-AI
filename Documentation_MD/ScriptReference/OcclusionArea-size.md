public [Vector3](Vector3.html) size;

### Description

Size that the occlusion area will have.

    
    
    using UnityEngine;  
      
    [[RequireComponent](RequireComponent.html)(typeof([OcclusionArea](OcclusionArea.html)))]
    public class Example : [MonoBehaviour](MonoBehaviour.html)
    {
        // Sets an Occlusion area of 10 units in x, 20 units in y and 30 units in z
        void Start()
        {
            transform.GetComponent<[OcclusionArea](OcclusionArea.html)>().size = new [Vector3](Vector3.html)(10, 20, 30);
        }
    }
