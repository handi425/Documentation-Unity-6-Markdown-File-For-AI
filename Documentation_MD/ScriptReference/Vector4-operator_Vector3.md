### Description

Converts a [Vector3](Vector3.html) to a Vector4.

Vector3s can be implicitly converted to Vector4 (w is set to zero in the
result).

    
    
    using UnityEngine;
    using System.Collections;  
      
    public class ExampleClass : [MonoBehaviour](MonoBehaviour.html)
    {
        void Start()
        {
            // color as a [Vector3](Vector3.html) (red, green, blue)
            [Vector3](Vector3.html) color = new [Vector3](Vector3.html)(1.0f, 0.25f, 0f);  
      
            // This converts the [Vector3](Vector3.html) to a [Vector4](Vector4.html) with the w parameter set to 0
            [Vector4](Vector4.html) colorWithOpacity = color;
            // set the opacity in the w parameter
            colorWithOpacity.w = 0.75f;
        }
    }
