public float depth;

### Description

Camera's depth in the camera rendering order.

Cameras with lower depth are rendered before cameras with higher depth.  
  
Use this to control the order in which cameras are drawn if you have multiple
cameras and some of them don't cover the full screen.  
  
Additional resources: [camera component](../Manual/class-Camera.html),
[Camera.rect](Camera-rect.html).

    
    
    using UnityEngine;  
      
    public class ExampleClass : [MonoBehaviour](MonoBehaviour.html)
    {
        [Camera](Camera.html) cam;  
      
        void Start()
        {
            // Set this camera to render after the main camera
            cam.depth = Camera.main.depth + 1;
        }
    }
