## Declaration

public static string NicifyVariableName(string name);

### Description

Make a displayable name for a variable.

This function will insert spaces before capital letters and remove optional
`m_`, `_` or `k` followed by uppercase letter in front of the name.

    
    
    using UnityEngine;
    using [UnityEditor](UnityEditor.html);  
      
    public class Example : [MonoBehaviour](MonoBehaviour.html)
    {
        private void Start()
        {
            // prints "My Variable"
            print([ObjectNames.NicifyVariableName](ObjectNames.NicifyVariableName.html)("MyVariable"));
            // prints "The Other Variable"
            print([ObjectNames.NicifyVariableName](ObjectNames.NicifyVariableName.html)("m_TheOtherVariable"));
            // prints "Some Constant"
            print([ObjectNames.NicifyVariableName](ObjectNames.NicifyVariableName.html)("kSomeConstant"));
        }
    }
