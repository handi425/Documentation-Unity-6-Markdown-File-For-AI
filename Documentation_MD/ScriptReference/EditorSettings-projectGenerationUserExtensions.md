public static string[] projectGenerationUserExtensions;

### Description

Controls list of extensions of files that will be included in the c# .csproj
projects that Unity generates.

Use this if you have certain file formats that you'd like to add to the
.csproj files for convenient editing. This setting takes a semicolon-separated
string of extensions. Leading dots are optional. This setting is specific to a
project.

    
    
    using UnityEngine;
    using [UnityEditor](UnityEditor.html);  
      
    public class Example
    {
        [[MenuItem](MenuItem.html)("TryMe/SetExtensions")]
        static void Doit()
        {
            [EditorSettings.projectGenerationUserExtensions](EditorSettings-projectGenerationUserExtensions.html) = new[] { "txt", "xml", ".cs", "myfunkyformat"};
        }
    }
